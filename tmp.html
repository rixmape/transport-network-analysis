import networkx as nx
import igraph as ig
import pandas as pd
import numpy as np


def create_graphs():
    # Use the Zachary Karate Club graph available in both libraries
    G_nx = nx.karate_club_graph()

    # Convert to igraph using adjacency list
    edges = list(G_nx.edges())
    G_ig = ig.Graph(edges=edges, directed=False)

    return G_nx, G_ig


def normalize_degree_igraph(degree, n):
    return [d / (n - 1) for d in degree]


def normalize_betweenness_igraph(betweenness, n):
    return [b / ((n - 1) * (n - 2)) for b in betweenness]


def compare_centralities():
    G_nx, G_ig = create_graphs()
    n = len(G_nx.nodes)

    # --- Compute NetworkX centralities ---
    nx_degree = nx.degree_centrality(G_nx)
    nx_closeness = nx.closeness_centrality(G_nx)
    nx_betweenness = nx.betweenness_centrality(G_nx, normalized=True)

    # --- Compute igraph centralities ---
    ig_degree_raw = G_ig.degree()
    ig_degree = normalize_degree_igraph(ig_degree_raw, n)
    ig_closeness = G_ig.closeness(normalized=True)
    ig_betweenness_raw = G_ig.betweenness()
    ig_betweenness = normalize_betweenness_igraph(ig_betweenness_raw, n)

    # --- Assemble comparison DataFrame ---
    df = pd.DataFrame(
        {
            "Node": list(range(n)),
            "NX_Degree": [nx_degree[i] for i in range(n)],
            "IG_Degree": ig_degree,
            "Degree_Diff": np.abs(np.array([nx_degree[i] for i in range(n)]) - np.array(ig_degree)),
            "NX_Closeness": [nx_closeness[i] for i in range(n)],
            "IG_Closeness": ig_closeness,
            "Closeness_Diff": np.abs(np.array([nx_closeness[i] for i in range(n)]) - np.array(ig_closeness)),
            "NX_Betweenness": [nx_betweenness[i] for i in range(n)],
            "IG_Betweenness": ig_betweenness,
            "Betweenness_Diff": np.abs(
                np.array([nx_betweenness[i] for i in range(n)]) - np.array(ig_betweenness)
            ),
        }
    )

    # --- Report ---
    print("Centrality Differences (NetworkX vs python-igraph):\n")
    print(df.round(4))

    # Check if differences are negligible
    tolerance = 1e-6
    summary = {
        "Degree match": all(df["Degree_Diff"] < tolerance),
        "Closeness match": all(df["Closeness_Diff"] < tolerance),
        "Betweenness match": all(df["Betweenness_Diff"] < tolerance),
    }
    print("\nSummary of matches within tolerance:")
    for k, v in summary.items():
        print(f"  {k}: {'✔️' if v else '❌'}")


if __name__ == "__main__":
    compare_centralities()
